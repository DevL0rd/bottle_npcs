cmake_minimum_required(VERSION 3.16)
project(bottle_npcs C CXX)

# A Dusklight native mod (detour-based). Builds into a single loadable module:
#   bottle_npcs.so (Linux) / .dll (Windows) / .dylib (macOS)
#
# It links against the game by symbol export, so it needs:
#   * the Dusklight source tree (game headers + the mod SDK header)
#   * the Dusklight build dir (generated + fetched headers, e.g. Tracy)
#   * funchook (runtime inline hooking)
# Override these paths with -D if your layout differs.

set(DUSKLIGHT_DIR "$ENV{HOME}/Github/dusklight"
    CACHE PATH "Dusklight source tree")
set(DUSKLIGHT_BUILD_DIR "${DUSKLIGHT_DIR}/build/linux-default-relwithdebinfo"
    CACHE PATH "Dusklight build directory (for generated + fetched headers)")
set(FUNCHOOK_DIR "$ENV{HOME}/Github/funchook"
    CACHE PATH "funchook source tree")

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_POSITION_INDEPENDENT_CODE ON)

# Build funchook as a static lib (no shared lib / tests / install).
set(FUNCHOOK_BUILD_SHARED OFF CACHE BOOL "" FORCE)
set(FUNCHOOK_BUILD_TESTS OFF CACHE BOOL "" FORCE)
set(FUNCHOOK_INSTALL OFF CACHE BOOL "" FORCE)
add_subdirectory(${FUNCHOOK_DIR} "${CMAKE_BINARY_DIR}/funchook")

add_library(bottle_npcs MODULE src/bottle_npcs.cpp)

# Assemble the ready-to-copy mod folder at build/<mod-id>/: the library plus
# mod.json, laid out exactly as it belongs in the game's mods/. (settings.json
# and config.json are generated by the mod/host at runtime, not shipped.)
# CMake's own scratch lives in staging/ -- see build.sh.
set(MOD_OUTPUT_DIR "${CMAKE_CURRENT_SOURCE_DIR}/build/${PROJECT_NAME}")
set_target_properties(bottle_npcs PROPERTIES
    PREFIX ""
    LIBRARY_OUTPUT_DIRECTORY "${MOD_OUTPUT_DIR}")
add_custom_command(TARGET bottle_npcs POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
        "${CMAKE_CURRENT_SOURCE_DIR}/src/mod.json" "${MOD_OUTPUT_DIR}/mod.json"
    COMMENT "Assembling mod folder -> build/${PROJECT_NAME}")

# Same compile defs the game is built with, so its headers parse identically.
target_compile_definitions(bottle_npcs PRIVATE
    TARGET_PC WIDESCREEN_SUPPORT=1 AVOID_UB=1 VERSION=0 MTX_USE_PS=1)

# Game header roots (mirror Dusklight's GAME_INCLUDE_DIRS) + funchook.
target_include_directories(bottle_npcs PRIVATE
    ${DUSKLIGHT_DIR}/include
    ${DUSKLIGHT_DIR}/src
    ${DUSKLIGHT_DIR}/assets/GZ2E01
    ${DUSKLIGHT_DIR}/libs/JSystem/include
    ${DUSKLIGHT_DIR}/libs
    ${DUSKLIGHT_DIR}/extern/aurora/include
    ${DUSKLIGHT_DIR}/extern/aurora/include/dolphin
    ${DUSKLIGHT_DIR}/extern
    ${DUSKLIGHT_BUILD_DIR}
    ${DUSKLIGHT_BUILD_DIR}/_deps/tracy-src/public
    ${FUNCHOOK_DIR}/include)

# Silence the game headers' own warnings; resolve game symbols at load time.
target_compile_options(bottle_npcs PRIVATE -w)
target_link_libraries(bottle_npcs PRIVATE funchook-static ${CMAKE_DL_LIBS})
